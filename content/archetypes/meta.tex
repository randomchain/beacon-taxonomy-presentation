\begin{frame}{Archetype: Autocratic Collector}
    \begin{columns}
        \begin{column}{0.6\textwidth}
            \begin{itemize}
                \item Collects private input
                \begin{itemize}
                    \item Random physical occurrences:\\{\scriptsize E.g.\ Background radiation, sun spots, weather, quantum-mechanical effects (photon splitters), \ldots}
                \end{itemize}
                \item Has no way of proving its honesty
            \end{itemize}
        \end{column}
        \begin{column}{0.4\textwidth}
            \begin{center}
                \scalebox{0.85}{
